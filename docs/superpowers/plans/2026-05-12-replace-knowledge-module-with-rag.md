# Replace coze-studio Knowledge Module with rag — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the in-tree Go knowledge module in coze-studio with the standalone `rag` Python/FastAPI service, called over HTTP. Keep coze's IDL contract intact; return HTTP 501 for features rag doesn't yet support.

**Architecture:** New `infra/rag` HTTP client + new `domain/knowledge/service/ragimpl` package implementing `service.Knowledge` by delegating to rag. Two-PR rollout: PR-1 lands new code behind a feature flag (`KNOWLEDGE_BACKEND=rag|legacy`, defaulting to `legacy`), PR-2 flips the default and deletes the legacy domain code. Mapping tables keep int64 coze IDs ↔ string rag UUIDs.

**Tech Stack:** Go 1.x (Hertz, gorm), FastAPI (rag), MySQL (coze metadata), MongoDB+ES+MinIO+Redis (rag), Atlas migrations, Vitest (frontend), Go testing.

**Spec:** `docs/superpowers/specs/2026-05-12-replace-knowledge-module-with-rag-design.md`

package rag

import (
	"fmt"
	"os"
	"time"

	"gopkg.in/yaml.v3"
)

type Config struct {
	BaseURL                       string        `yaml:"base_url"`
	Timeout                       time.Duration `yaml:"-"`
	TimeoutMs                     int           `yaml:"timeout_ms"`
	UploadTimeoutMs               int           `yaml:"upload_timeout_ms"`
	RetrievalTimeoutMs            int           `yaml:"retrieval_timeout_ms"`
	MaxRetries                    int           `yaml:"max_retries"`
	RetryBackoffMs                int           `yaml:"retry_backoff_ms"`
	DefaultTextEmbeddingModelID   string        `yaml:"default_text_embedding_model_id"`
	DefaultImageEmbeddingModelID  string        `yaml:"default_image_embedding_model_id"`
}

type FileConfig struct {
	Rag       Config           `yaml:"rag"`
	Knowledge KnowledgeBackend `yaml:"knowledge"`
}

type KnowledgeBackend struct {
	Backend string       `yaml:"backend"`
	Tenant  TenantConfig `yaml:"tenant"`
}

type TenantConfig struct {
	Mode            string `yaml:"mode"`              // "env" (Phase 1) | "user" (Phase 2)
	DefaultTenantID string `yaml:"default_tenant_id"` // used when Mode=="env"
}

func Load(path string) (*FileConfig, error) {
	b, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("read rag config: %w", err)
	}
	expanded := os.ExpandEnv(string(b))
	var c FileConfig
	if err := yaml.Unmarshal([]byte(expanded), &c); err != nil {
		return nil, fmt.Errorf("parse rag config: %w", err)
	}
	c.Rag.Timeout = time.Duration(c.Rag.TimeoutMs) * time.Millisecond
	return &c, nil
}
```

package rag

import "context"

// Client is the abstraction over the rag HTTP service that ragimpl depends on.
// All methods are tenant-scoped via the request bodies; the underlying HTTP
// client never invents a tenant_id.
//
//go:generate mockgen -destination ../../../internal/mock/infra/rag/client_mock.go -package mock -source client.go Client
type Client interface {
	// Health
	Ready(ctx context.Context) error

	// Model providers (used by the create-KB proxy)
	ListModelProviders(ctx context.Context) (*ListModelProvidersResponse, error)

	// Knowledge bases
	CreateKB(ctx context.Context, req *CreateKBRequest) (*KB, error)
	GetKB(ctx context.Context, tenantID, kbID string) (*KB, error)
	UpdateKB(ctx context.Context, tenantID, kbID string, req *UpdateKBRequest) (*KB, error)
	DeleteKB(ctx context.Context, tenantID, kbID string) error
	ListKBs(ctx context.Context, req *ListKBsRequest) (*ListKBsResponse, error)

	// Documents
	CreateDocument(ctx context.Context, kbID string, req *CreateDocumentRequest) (*CreateDocumentResponse, error)
	GetDocument(ctx context.Context, tenantID, docID string) (*Document, error)
	ListDocuments(ctx context.Context, tenantID, kbID string, page, pageSize int) (*ListDocumentsResponse, error)
	DeleteDocument(ctx context.Context, tenantID, docID string) error

package rag

import (
	"fmt"

	"github.com/coze-dev/coze-studio/backend/pkg/errorx"
	"github.com/coze-dev/coze-studio/backend/types/errno"
)

// MapRagError converts a rag-side error code to a coze errno.
// rag codes follow the layout in 总体设计文档.md §12.6:
//   40001-40009  → invalid param / capability mismatch
//   404xx        → not found
//   409xx        → conflict
//   500xx        → upstream internal
func MapRagError(httpStatus, ragCode int, ragMessage string) error {
	switch {
	case ragCode >= 40001 && ragCode <= 40009:
		return errorx.New(errno.ErrKnowledgeInvalidParamCode, errorx.KV("detail", fmt.Sprintf("rag %d: %s", ragCode, ragMessage)))
	case ragCode >= 40400 && ragCode < 40500:
		// 40401 KB, 40402 doc, etc. — best-effort distinction by code suffix.
		if ragCode == 40401 {
			return errorx.New(errno.ErrKnowledgeNotExistCode, errorx.KV("detail", ragMessage))
		}
		if ragCode == 40402 {
			return errorx.New(errno.ErrKnowledgeDocumentNotExistCode, errorx.KV("detail", ragMessage))
		}
		return errorx.New(errno.ErrKnowledgeNotExistCode, errorx.KV("detail", fmt.Sprintf("rag %d: %s", ragCode, ragMessage)))
	case ragCode >= 40900 && ragCode < 41000:
		return errorx.New(errno.ErrKnowledgeDuplicateCode, errorx.KV("detail", fmt.Sprintf("rag %d: %s", ragCode, ragMessage)))
	default:
		// Fallback: treat as upstream unavailable.
		return errorx.New(errno.ErrRagUpstreamUnavailableCode, errorx.KV("detail", fmt.Sprintf("http=%d rag=%d msg=%s", httpStatus, ragCode, ragMessage)))
	}
}

package rag

import (
	"bytes"
	"context"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"time"

	"github.com/coze-dev/coze-studio/backend/conf/rag"

)

// Compile-time check that *Client satisfies the contract.
var _ contract.Client = (*Client)(nil)

type Client struct {
	cfg  rag.Config
	http *http.Client
}

func New(cfg rag.Config) *Client {
	return &Client{
		cfg: cfg,
		http: &http.Client{
			Timeout: cfg.Timeout,
		},
	}
}

func (c *Client) Ready(ctx context.Context) error {
	return c.doJSON(ctx, http.MethodGet, "/ready", nil, nil, c.cfg.Timeout)
}

// doJSON executes a JSON-in/JSON-out request. If body == nil no request body is sent.
// If out == nil the response body is discarded. Honors retry policy for idempotent methods only.
func (c *Client) doJSON(ctx context.Context, method, path string, body, out any, timeout time.Duration) error {
	var lastErr error
	attempts := 1 + c.cfg.MaxRetries
	for attempt := 0; attempt < attempts; attempt++ {
		err := c.doOnce(ctx, method, path, body, out, timeout)
		if err == nil {
			return nil
		}
		lastErr = err
		if !idempotent(method) {
			return err
		}
		// Backoff with exponential growth; capped.
		select {
		case <-ctx.Done():
			return ctx.Err()
		case <-time.After(time.Duration((attempt+1)*c.cfg.RetryBackoffMs) * time.Millisecond):
		}
	}
	return lastErr
}

func (c *Client) doOnce(ctx context.Context, method, path string, body, out any, timeout time.Duration) error {
	var reader io.Reader
	if body != nil {
		buf, err := json.Marshal(body)
		if err != nil {
			return fmt.Errorf("marshal body: %w", err)
		}
		reader = bytes.NewReader(buf)
	}

	reqCtx, cancel := context.WithTimeout(ctx, timeout)
	defer cancel()

	req, err := http.NewRequestWithContext(reqCtx, method, c.cfg.BaseURL+path, reader)
	if err != nil {
		return fmt.Errorf("new request: %w", err)
	}
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("Accept", "application/json")
	// No Authorization header: rag has no service-to-service auth in current scope.
	// See spec §11 risk #1. If rag ever adds a token-check middleware, inject here.

	resp, err := c.http.Do(req)
	if err != nil {
		return fmt.Errorf("rag http %s %s: %w", method, path, err)
	}
	defer resp.Body.Close()

	if resp.StatusCode >= 400 {
		raw, _ := io.ReadAll(resp.Body)
		var errBody contract.ErrorBody
		_ = json.Unmarshal(raw, &errBody)
		return MapRagError(resp.StatusCode, errBody.Code, errBody.Message)
	}

	if out == nil {
		return nil
	}
	if err := json.NewDecoder(resp.Body).Decode(out); err != nil {
		return fmt.Errorf("decode response: %w", err)
	}
	return nil
}

func idempotent(method string) bool {
	return method == http.MethodGet || method == http.MethodDelete
}

func (m *MappingRepo) KBByCozeID(ctx context.Context, cozeID int64) (*KBMapping, error) {
	var row struct {
		CozeKBID  int64  `gorm:"column:coze_kb_id"`
		RagKBID   string `gorm:"column:rag_kb_id"`
		IconURI   string `gorm:"column:icon_uri"`

	}, nil

// KBsByCozeIDs returns mappings for the requested ids. It does NOT enforce any
// tenant invariant — tenant is resolved by TenantResolver at the call site and
// rag itself rejects cross-tenant access by filtering on tenant_id.
func (m *MappingRepo) KBsByCozeIDs(ctx context.Context, ids []int64) ([]*KBMapping, error) {
	if len(ids) == 0 {
		return nil, nil
	}
	var rows []struct {
		CozeKBID  int64  `gorm:"column:coze_kb_id"`
		RagKBID   string `gorm:"column:rag_kb_id"`

		Select("coze_kb_id, rag_kb_id, icon_uri, app_id, creator_id").
		Where("coze_kb_id IN ? AND (deleted_at IS NULL)", ids).
		Scan(&rows).Error

		return nil, err

			AppID: r.AppID, CreatorID: r.CreatorID,

		Name:                      req.Name,
		Description:               req.Description,

		ImageEmbeddingModelID:     imageModel,
		EnabledChunkTypes:         defaultChunkTypesFor(req.FormatType),
		SupportedSourceModalities: defaultSourceModalitiesFor(req.FormatType),
		DefaultFusionPolicy: contract.FusionPolicy{
			Mode: "weighted_rrf", RrfK: 60,

	// not persisted here — rag is the authoritative source.
	if err := i.mapping.InsertKB(ctx, cozeID, ragResp.KBID, req.IconUri, req.AppID, req.CreatorID, now); err != nil {
		// Best-effort rollback: ask rag to delete the KB it just made.
		_ = i.rag.DeleteKB(ctx, tenant, ragResp.KBID)
		return nil, fmt.Errorf("insert mapping: %w", err)
	}
	return &service.CreateKnowledgeResponse{KnowledgeID: cozeID, CreatedAtMs: now}, nil

			progress = append(progress, &service.DocumentProgress{ID: id, Progress: 100, Status: entity.DocumentStatusEnable})
			continue
		}
		t, err := i.rag.GetTask(ctx, tenant, m.LastTaskID)
		if err != nil {
			continue
		}
		// Note: we do NOT mirror status to MySQL — the mapping table no longer
		// has a `status` column. Each progress call reads rag live.
		progress = append(progress, &service.DocumentProgress{
			ID: id, Progress: t.Progress, Status: taskStatusToDoc(t.Status), StatusMsg: t.Error,
		})
	}
	return &service.MGetDocumentProgressResponse{ProgressList: progress}, nil
}

func sourceModalityFor(d *entity.Document) string {
	// In v1: text_source by default; image_source if doc is image type.
	if d.Type == entity.DocumentTypeImage {
		return "image_source"
	}
	return "text_source"
}

func buildDocMetadata(d *entity.Document) map[string]any {
	return map[string]any{
		"coze_document_id": d.ID,
		"creator_id":       d.CreatorID,
	}
}

func taskStatusToDoc(s string) entity.DocumentStatus {
	switch s {
	case "success":
		return entity.DocumentStatusEnable
	case "failed":
		return entity.DocumentStatusFailed
	case "pending", "retrying":
		return entity.DocumentStatusProcessing
	case "running":
		return entity.DocumentStatusProcessing
	default:
		return entity.DocumentStatusFailed
	}

package ragimpl

import (
	"context"
	"errors"
	"fmt"

	contract "github.com/coze-dev/coze-studio/backend/infra/contract/rag"
	knowledgeModel "github.com/coze-dev/coze-studio/backend/crossdomain/knowledge/model"
	"github.com/coze-dev/coze-studio/backend/domain/knowledge/entity"
	"github.com/coze-dev/coze-studio/backend/pkg/errorx"
	"github.com/coze-dev/coze-studio/backend/types/errno"
)

func (i *Impl) Retrieve(ctx context.Context, req *knowledgeModel.RetrieveRequest) (*knowledgeModel.RetrieveResponse, error) {
	// 1. NL2SQL sub-feature guard (Retrieve is bucket A overall but NL2SQL is bucket B).
	if req.Strategy != nil && req.Strategy.EnableNL2SQL {
		return nil, errorx.New(errno.ErrRagFeaturePendingCode, errorx.KV("detail",
			"NL2SQL retrieval is pending rag support (roadmap: rag/docs/notes/roadmap.md#nl2sql)"))
	}

	if len(req.KnowledgeIDs) == 0 {
		return nil, errors.New("at least one knowledge_id required")
	}

	// 2. Resolve KB mappings (no coze-side tenant check; rag enforces).
	kbs, err := i.mapping.KBsByCozeIDs(ctx, req.KnowledgeIDs)
	if err != nil {
		return nil, err
	}
	if len(kbs) == 0 {
		return nil, errors.New("no kbs resolved")
	}
	tenant, err := i.tenant(ctx)
	if err != nil {
		return nil, err
	}
	ragKBIDs := make([]string, 0, len(kbs))
	for _, k := range kbs {
		ragKBIDs = append(ragKBIDs, k.RagKBID)
	}

	// 3. Translate doc IDs if provided.
	var ragDocIDs []string
	if len(req.DocumentIDs) > 0 {
		docs, err := i.mapping.DocsByCozeIDs(ctx, req.DocumentIDs)
		if err != nil {
			return nil, err
		}
		for _, d := range docs {
			ragDocIDs = append(ragDocIDs, d.RagDocID)
		}
	}

	// 4. Build rag request.
	ragReq := &contract.RetrieveRequest{
		TenantID:  tenant,
		KBIDs:     ragKBIDs,
		DocIDs:    ragDocIDs,
		Query:     req.Query,
		QueryMode: "text_input",
	}
	if req.Strategy != nil {
		if req.Strategy.TopK != nil {
			ragReq.TopK = int(*req.Strategy.TopK)
		}
		if req.Strategy.MinScore != nil {
			ragReq.MinScore = *req.Strategy.MinScore
		}
		if req.Strategy.MaxTokens != nil {
			ragReq.MaxTokens = int(*req.Strategy.MaxTokens)
		}
		switch req.Strategy.SearchType {
		case knowledgeModel.SearchTypeFullText:
			ragReq.SearchType = "fulltext"
		case knowledgeModel.SearchTypeHybrid:
			ragReq.SearchType = "hybrid"
		default:
			ragReq.SearchType = "semantic"
		}
		if req.Strategy.EnableQueryRewrite {
			ragReq.QueryStrat = map[string]any{"rewrite": true}
		}
		if req.Strategy.EnableRerank {
			ragReq.Rerank = map[string]any{"enabled": true}
		}
	}

	// 5. Call rag.
	resp, err := i.rag.Retrieve(ctx, ragReq)
	if err != nil {
		return nil, err
	}

	// 6. Translate hits back to coze entities.
	slices := make([]*knowledgeModel.RetrieveSlice, 0, len(resp.Hits))
	for _, h := range resp.Hits {
		m, err := i.mapping.docByRagID(ctx, h.DocID)
		if err != nil {
			// Skip hits we can't map back (e.g. stale rag data); log in production.
			continue
		}
		s := &knowledgeModel.Slice{
			Info:        knowledgeModel.Info{ID: 0}, // chunk-level ID not exposed in v1
			KnowledgeID: m.KBID,
			DocumentID:  m.CozeID,
			RawContent:  []*knowledgeModel.SliceContent{{Type: knowledgeModel.SliceContentTypeText, Text: ptrStr(h.Content)}},
			SliceStatus: knowledgeModel.SliceStatus(0),
		}
		slices = append(slices, &knowledgeModel.RetrieveSlice{Slice: s, Score: h.Score})
	}
	return &knowledgeModel.RetrieveResponse{RetrieveSlices: slices}, nil
}

func ptrStr(s string) *string {
	return &s
}

		domainSVC, eventHandler = knowledgeImpl.NewKnowledgeSVC(c)

		}
	}

	KnowledgeSVC.DomainSVC = domainSVC
	KnowledgeSVC.eventBus = bus
	KnowledgeSVC.storage = c.Storage
	return KnowledgeSVC, nil
}

// rag-contract-check: fetch rag's /openapi.json and assert that the endpoints
// coze's Go client expects are still present with the expected method.
package main

import (
	"encoding/json"
	"flag"
	"fmt"
	"net/http"
	"os"
)

type openAPI struct {
	Paths map[string]map[string]any `json:"paths"`
}

var required = []struct {
	Path   string
	Method string
}{
	{"/ready", "get"},
	{"/model_providers", "get"},
	{"/knowledgebases", "post"},
	{"/knowledgebases", "get"},
	{"/knowledgebases/{kb_id}", "get"},
	{"/knowledgebases/{kb_id}", "patch"},
	{"/knowledgebases/{kb_id}", "delete"},
	{"/knowledgebases/{kb_id}/documents", "post"},
	{"/knowledgebases/{kb_id}/documents", "get"},
	{"/documents/{doc_id}", "get"},
	{"/documents/{doc_id}", "delete"},
	{"/tasks/{task_id}", "get"},
	{"/retrieval", "post"},
}

func main() {
	base := flag.String("base", "http://localhost:8000", "rag base URL")
	flag.Parse()

	resp, err := http.Get(*base + "/openapi.json")
	if err != nil {
		fmt.Fprintln(os.Stderr, "fetch:", err)
		os.Exit(2)
	}
	defer resp.Body.Close()
	var oa openAPI
	if err := json.NewDecoder(resp.Body).Decode(&oa); err != nil {
		fmt.Fprintln(os.Stderr, "decode:", err)
		os.Exit(2)
	}
	missing := 0
	for _, r := range required {
		methods, ok := oa.Paths[r.Path]
		if !ok {
			fmt.Fprintf(os.Stderr, "MISSING path: %s\n", r.Path)
			missing++
			continue
		}
		if _, ok := methods[r.Method]; !ok {
			fmt.Fprintf(os.Stderr, "MISSING method %s on path %s\n", r.Method, r.Path)
			missing++
		}
	}
	if missing > 0 {
		fmt.Fprintf(os.Stderr, "%d contract violations\n", missing)
		os.Exit(1)
	}
	fmt.Println("OK")
}
```
